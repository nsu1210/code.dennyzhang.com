Given a collection of intervals, merge all overlapping intervals.

#+BEGIN_EXAMPLE
For example,
Given [1,3],[2,6],[8,10],[15,18],
return [1,6],[8,10],[15,18].
#+END_EXAMPLE

## https://code.dennyzhang.com/merge-intervals
## Basic Ideas: sorting
##
##  Sort by starting point. If a tie, then by ending point
##  The first one would be first interval
##
## Complexity: Time O(n*log(n)), Space O(n)
class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        # sort by starting point first, then ending point
        intervals.sort()
        res = []
        for v in intervals:
            [s, e] = v
            # overlapping
            if res and res[-1][1] >= s:
                res[-1][1] = max(res[-1][1], e)
            else:
                res.append(v)
        return res
#+END_SRC
